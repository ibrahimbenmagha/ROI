

Table users {
  id integer [primary key] 
  FirstName varchar
  LastName varchar
  role varchar
  Email email
  PSW pasword
  created_at timestamp
}

Table Labo {
  id integer [primary key] 
  userId integer
  Status boolean
  Name varchar
  created_at timestamp 
}

Table admins {
  
  id integer [primary key]
  userId integer
}

  Name varchar
}

Table ActivityByLabo{

  laboId integer

}

Table ActivityItems{

  Name varchar
  ActivityId integer
}

Table ActivityItemsValue{
